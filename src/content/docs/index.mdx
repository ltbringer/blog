import { Card, CardGrid, Badge } from '@astrojs/starlight/components';
import { getCollection } from 'astro:content';

export const allDocs = await getCollection('docs');

export const blogs = allDocs
	.filter(doc => doc.id !== 'index.mdx')
	.map(doc => {
		// Extract date from folder path (e.g., "2025-12-13/article.mdx")
		const dateMatch = doc.id.match(/^(\d{4}-\d{2}-\d{2})\//);
		const dateStr = dateMatch ? dateMatch[1] : null;
		const articleDate = dateStr ? new Date(dateStr) : null;

		// Check if article is less than a week old
		const now = new Date();
		const weekAgo = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000);
		const isNew = articleDate && articleDate > weekAgo;

		return {
			...doc,
			articleDate,
			isNew,
			slug: doc.id,
		};
	})
	.sort((a, b) => {
		// Sort by date descending (newest first)
		if (!a.articleDate) return 1;
		if (!b.articleDate) return -1;
		return b.articleDate.getTime() - a.articleDate.getTime();
	});
